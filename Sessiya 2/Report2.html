def preprocess_text(data, stopwords=russian_stopwords):
    text = re.sub('','', data.lower())
    text = text.strip()
    text = [w for w in text.split() if w not in stopwords] 
    text = [w for w in text if len(w) >=3]
    return ' '.join(text)
